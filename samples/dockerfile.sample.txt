# syntax=docker/dockerfile:1

# This is a sample Dockerfile for using the Artifacts Credential Provider.
# Build with: $env:PAT="<your-pat>"; docker build --secret id=pat,env=PAT .

# downloading the dotnet sdk image. Could be any docker sdk image with sdk > 9.0.2 
# With nuget plugin dotnet tool install support https://learn.microsoft.com/nuget/release-notes/nuget-6.13#summary-whats-new-in-6130)
FROM mcr.microsoft.com/dotnet/sdk:10.0 AS dotnet-builder

# Add dotnet tools to path
ENV PATH="$PATH:/root/.dotnet/tools"

# download and install latest credential provider.
# use --version X.* to pin to a specific major version.
RUN dotnet tool install --global  Microsoft.Artifacts.CredentialProvider.NuGet.Tool --source https://api.nuget.org/v3/index.json

# Optional
WORKDIR /workdir
COPY ./ .

# Optional: Sometimes the http client hangs because of a .NET issue. Setting this in dockerfile helps 
ENV DOTNET_SYSTEM_NET_HTTP_USESOCKETSHTTPHANDLER=0

# Optional: Disable file caches as a precaution to ensure no token caches persist in the image.
# When using environment variables (e.g. ARTIFACTS_CREDENTIALPROVIDER_ACCESSTOKEN) the credential provider
# should not write secrets to these caches, but disabling them provides an extra safeguard.
ENV ARTIFACTS_CREDENTIALPROVIDER_SESSIONTOKENCACHE_ENABLED=false
ENV ARTIFACTS_CREDENTIALPROVIDER_MSAL_FILECACHE_ENABLED=false

# Use a secret mount to pass the PAT securely during restore. The secret is never persisted in any image layer.
# ARTIFACTS_CREDENTIALPROVIDER_URI_PREFIXES: semicolon-separated list of feed URI prefixes to authenticate.
# ARTIFACTS_CREDENTIALPROVIDER_ACCESSTOKEN: the PAT used for authentication.
RUN --mount=type=secret,id=pat \
  export ARTIFACTS_CREDENTIALPROVIDER_URI_PREFIXES="https://pkgs.dev.azure.com" && \
  export ARTIFACTS_CREDENTIALPROVIDER_ACCESSTOKEN="$(cat /run/secrets/pat)" && \
  dotnet restore

# Optional: Extended step to build the app using dotnet msbuild. 
RUN dotnet build dirs.proj
